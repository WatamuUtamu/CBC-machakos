  <meta name="viewport" content="width=device-width, initial-scale=1.0">

  <link rel="manifest" href="manifest.json">
  <script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
  <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
  <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
  <script src="https://cdn.tailwindcss.com"></script>
  <style>
    body { background: linear-gradient(135deg, #f0fdf4 0%, #eef2ff 100%); min-height: 100vh; }
    .glass { background: rgba(255, 255, 255, 0.9); backdrop-filter: blur(10px); border-radius: 2rem; }
  </style>
</head>
<body>
  <div id="root"></div>
  <script type="text/babel">
    const { useState, useEffect } = React;

    function App() {
      const [step, setStep] = useState(1);
      const [grade, setGrade] = useState('');
      
      useEffect(() => {
        if ('serviceWorker' in navigator) {
          navigator.serviceWorker.register('./sw.js');
        }
      }, []);

      return (
        <div className="max-w-md mx-auto p-6 text-center">
          <h1 className="text-3xl font-black text-green-700 mt-10">CBC Machakos</h1>
          <p className="mb-8 text-slate-500">Kangundo • Tala • Koma Rock</p>
          
          {step === 1 && (
            <div className="glass p-8 shadow-xl">
              <h2 className="text-xl font-bold mb-4">Select Your Grade</h2>
              {['Grade 7', 'Grade 8', 'Grade 9'].map(g => (
                <button key={g} onClick={() => {setGrade(g); setStep(2);}} 
                className="w-full p-4 mb-3 bg-white border border-slate-200 rounded-xl font-bold hover:bg-green-50">
                  {g}
                </button>
              ))}
            </div>
          )}

          {step === 2 && (
            <div className="glass p-8 shadow-xl">
              <h2 className="text-2xl font-black text-slate-800">Your {grade} Pathway</h2>
              <p className="mt-4 text-slate-600">Local Machakos TVETs and Senior School tracks will appear here.</p>
              <button onClick={() => setStep(1)} className="mt-6 text-green-600 font-bold">← Go Back</button>
            </div>
          )}
        </div>
      );
    }
    ReactDOM.createRoot(document.getElementById('root')).render(<App />);
  </script>
</body>
</html>
